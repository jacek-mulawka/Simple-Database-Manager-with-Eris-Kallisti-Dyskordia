select  ALL_CONS_COLUMNS.TABLE_NAME
     , ALL_CONS_COLUMNS.COLUMN_NAME as PRIMARY_KEY_NAME
     
     , ALL_CONS_COLUMNS.POSITION
     , ALL_CONSTRAINTS.STATUS
     , ALL_CONSTRAINTS.OWNER as OWNER_NAME
     
from  ALL_CONSTRAINTS, ALL_CONS_COLUMNS
where  ALL_CONS_COLUMNS.TABLE_NAME = :table_name
  and ALL_CONSTRAINTS.CONSTRAINT_TYPE = 'P'
  and ALL_CONSTRAINTS.CONSTRAINT_NAME = ALL_CONS_COLUMNS.CONSTRAINT_NAME
and ALL_CONSTRAINTS.OWNER = ALL_CONS_COLUMNS.OWNER
