select SYS.DBA_TAB_COLUMNS.COLUMN_ID as ID_FIELD_SOURCE_NAME


     , SYS.DBA_TAB_COLUMNS.COLLATION as BASE_COLLATION_NAME
     , '<?>' as COLLATION_SCHEMA
     , SYS.DBA_TAB_COLUMNS.CHARACTER_SET_NAME as CHARACTER_SET_NAME
     , SYS.DBA_TAB_COLUMNS.COLLATION as COLLATION_NAME
     , SYS.DBA_TAB_COLUMNS.COLUMN_NAME as COLUMN_NAME
     , SYS.DBA_TAB_COLUMNS.DATA_DEFAULT as DEFAULT_VALUE__CAST
     , SYS.DBA_TAB_COLUMNS.DATA_DEFAULT as DEFAULT_VALUE
     , '<?>' as DESCRIPTION_VALUE
/*     , iif( SYS.DBA_TAB_COLUMNS.DATA_PRECISION is not null, SYS.DBA_TAB_COLUMNS.DATA_LENGTH, SYS.DBA_TAB_COLUMNS.CHAR_LENGTH ) as FIELD_LENGTH*/
     , (case
         when ( SYS.DBA_TAB_COLUMNS.DATA_PRECISION is not null ) then
           SYS.DBA_TAB_COLUMNS.DATA_LENGTH
         else
           SYS.DBA_TAB_COLUMNS.CHAR_LENGTH
       end) as FIELD_LENGTH
/*     , iif( SYS.DBA_TAB_COLUMNS.DATA_PRECISION is not null, SYS.DBA_TAB_COLUMNS.DATA_SCALE, SYS.DBA_TAB_COLUMNS.CHAR_COL_DECL_LENGTH ) as FIELD_SCALE*/
     , (case
         when ( SYS.DBA_TAB_COLUMNS.DATA_PRECISION is not null ) then
           SYS.DBA_TAB_COLUMNS.DATA_SCALE
         else
           SYS.DBA_TAB_COLUMNS.CHAR_COL_DECL_LENGTH
       end) as FIELD_SCALE   

     , SYS.DBA_TAB_COLUMNS.DATA_TYPE as FIELD_TYPE

/*     , iif( SYS.DBA_TAB_COLUMNS.NULLABLE = 'N', 1, null ) as NOT_NULL*/
     , (case
         when ( SYS.DBA_TAB_COLUMNS.NULLABLE = 'N' ) then
           1
         else
           null
       end) as NOT_NULL

     , case when SYS.DBA_TAB_COLUMNS.NULLABLE = 'N' then 1 else 2 end

     , SYS.DBA_TAB_COLUMNS.OWNER as OWNER_NAME
     , '<?>' as SEGMENT_LENGTH

     , '<?>' as METADATA_LOG

from SYS.DBA_TAB_COLUMNS
where SYS.DBA_TAB_COLUMNS.TABLE_NAME = :table_name
order by SYS.DBA_TAB_COLUMNS.COLUMN_ID
